#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ImageAnalysis {
    pub width: u32,
    pub height: u32,
    pub format: ImageFormat,
    pub aspect_ratio: f32,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum ImageFormat {
    Jpeg,
    Png,
    Gif,
    Webp,
    Bmp,
    Svg,
    Other(String),
}

impl From<&str> for ImageFormat {
    fn from(ext: &str) -> Self {
        match ext.to_lowercase().as_str() {
            "jpg" | "jpeg" => ImageFormat::Jpeg,
            "png" => ImageFormat::Png,
            "gif" => ImageFormat::Gif,
            "webp" => ImageFormat::Webp,
            "bmp" => ImageFormat::Bmp,
            "svg" => ImageFormat::Svg,
            other => ImageFormat::Other(other.to_string()),
        }
    }
}

pub fn analyze_image(path: &Path) -> Result<ImageAnalysis, String> {
    let ext = path.extension()
        .and_then(|e| e.to_str())
        .unwrap_or("");
    
    // For SVG, we need special handling
    if ext.eq_ignore_ascii_case("svg") {
        // Try to extract viewBox or width/height attributes
        // For now, return placeholder dimensions
        return Ok(ImageAnalysis {
            width: 512,
            height: 512,
            format: ImageFormat::Svg,
            aspect_ratio: 1.0,
        });
    }
    
    let img = image::open(path).map_err(|e| e.to_string())?;
    let (width, height) = (img.width(), img.height());
    
    Ok(ImageAnalysis {
        width,
        height,
        format: ImageFormat::from(ext),
        aspect_ratio: width as f32 / height as f32,
    })
}

pub fn get_image_dimensions(path: &Path) -> Result<(u32, u32), String> {
    analyze_image(path).map(|a| (a.width, a.height))
}

use image::{DynamicImage, GenericImageView, Rgba};
use serde::{Deserialize, Serialize};
use std::path::Path;

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Color {
    pub r: u8,
    pub g: u8,
    pub b: u8,
    pub percentage: f32,
}

impl Color {
    pub fn to_hex(&self) -> String {
        format!("#{:02X}{:02X}{:02X}", self.r, self.g, self.b)
    }
    
    pub fn to_rgb_string(&self) -> String {
        format!("rgb({}, {}, {})", self.r, self.g, self.b)
    }
}

pub fn extract_dominant_colors(
    path: &Path,
    color_count: u8,
) -> Result<Vec<Color>, String> {
    let img = image::open(path).map_err(|e| e.to_string())?;
    
    // Resize for faster processing (maintain aspect ratio, max dimension 150)
    let thumb = img.thumbnail(150, 150);
    
    // Sample pixels (every 4th pixel for performance)
    let mut pixels: Vec<[f32; 3]> = Vec::new();
    for (x, y, pixel) in thumb.pixels() {
        if (x + y) % 4 == 0 { // Sample every 4th pixel
            let Rgba([r, g, b, a]) = pixel;
            // Skip transparent pixels
            if a > 128 {
                pixels.push([r as f32, g as f32, b as f32]);
            }
        }
    }
    
    if pixels.is_empty() {
        return Ok(vec![]);
    }
    
    // Run k-means clustering
    let k = color_count.min(10).max(1) as usize;
    let kmeans = kmeans::KMeans::new(pixels, pixels.len(), 3);
    let result = kmeans.kmeans_lloyd(k, 100, kmeans::KMeans::init_random_sample);
    
    // Build color results with percentages
    let mut colors: Vec<(Color, usize)> = Vec::new();
    for (i, centroid) in result.centroids.iter().enumerate() {
        let cluster_size = result.assignments.iter().filter(|&&a| a == i).count();
        let percentage = (cluster_size as f32 / pixels.len() as f32) * 100.0;
        
        colors.push((Color {
            r: centroid[0] as u8,
            g: centroid[1] as u8,
            b: centroid[2] as u8,
            percentage,
        }, cluster_size));
    }
    
    // Sort by cluster size (descending) and take top colors
    colors.sort_by(|a, b| b.1.cmp(&a.1));
    colors.truncate(color_count as usize);
    
    Ok(colors.into_iter().map(|(c, _)| c).collect())
}

// Sync version for spawn_blocking
pub fn extract_dominant_colors_sync(
    path: &Path,
    color_count: u8,
) -> Result<Vec<Color>, String> {
    extract_dominant_colors(path, color_count)
}

    db: State<'_, Arc<Database>>,

    // Check cache first

        // Parse cached analysis

            cached: true,

        .map_err(|e| e.to_string())?
        .ok_or_else(|| "File not found".to_string())?;

// Cache image analysis
pub async fn cache_image_analysis(
    &self,
    file_id: i64,
    analysis: &ImageAnalysis,
) -> Result<(), sqlx::Error> {
    let json = serde_json::to_string(analysis).unwrap_or_default();
    sqlx::query(
        "UPDATE files SET metadata = COALESCE(metadata, '{}') || ? WHERE id = ?"
    )
    .bind(format!('{{"image_analysis": {}}}', json))
    .bind(file_id)
    .execute(&self.pool)
    .await?;
    Ok(())
}

pub async fn get_image_analysis(
    &self,
    file_id: i64,
) -> Result<Option<ImageAnalysis>, sqlx::Error> {
    // Query metadata JSON for cached analysis
    let row: Option<(String,)> = sqlx::query_as(
        "SELECT metadata FROM files WHERE id = ? AND metadata IS NOT NULL"
    )
    .bind(file_id)
    .fetch_optional(&self.pool)
    .await?;
    
    if let Some((metadata_json,)) = row {
        if let Ok(metadata) = serde_json::from_str::<serde_json::Value>(&metadata_json) {
            if let Some(analysis) = metadata.get("image_analysis") {
                return Ok(serde_json::from_value(analysis).ok());
            }
        }
    }
    Ok(None)
}

import { useState, useEffect, useCallback, useMemo, useRef } from 'react';

interface VirtualGridOptions {
  itemCount: number;
  itemWidth: number;
  itemHeight: number;
  gap: number;
  overscan?: number;
}

interface VirtualGridState {
  startIndex: number;
  endIndex: number;
  visibleItems: number[];
  totalHeight: number;
  scrollTop: number;
}

export function useVirtualGrid(options: VirtualGridOptions) {
  const { itemCount, itemWidth, itemHeight, gap, overscan = 2 } = options;
  const containerRef = useRef<HTMLDivElement>(null);
  const [scrollTop, setScrollTop] = useState(0);
  const [containerWidth, setContainerWidth] = useState(0);

  // Calculate columns based on container width
  const columns = useMemo(() => {
    if (containerWidth === 0) return 1;
    return Math.max(1, Math.floor((containerWidth + gap) / (itemWidth + gap)));
  }, [containerWidth, itemWidth, gap]);

  // Calculate total rows
  const rows = useMemo(() => {
    return Math.ceil(itemCount / columns);
  }, [itemCount, columns]);

  // Calculate total height
  const totalHeight = useMemo(() => {
    return rows * (itemHeight + gap) - gap;
  }, [rows, itemHeight, gap]);

  // Calculate visible range
  const visibleRange = useMemo(() => {
    const containerHeight = containerRef.current?.clientHeight || 0;
    
    // Calculate which rows are visible
    const startRow = Math.max(0, Math.floor(scrollTop / (itemHeight + gap)) - overscan);
    const endRow = Math.min(
      rows - 1,
      Math.ceil((scrollTop + containerHeight) / (itemHeight + gap)) + overscan
    );
    
    const startIndex = startRow * columns;
    const endIndex = Math.min(itemCount - 1, (endRow + 1) * columns - 1);
    
    // Generate array of visible item indices
    const visibleItems: number[] = [];
    for (let i = startIndex; i <= endIndex; i++) {
      visibleItems.push(i);
    }
    
    return {
      startIndex,
      endIndex,
      visibleItems,
    };
  }, [scrollTop, itemHeight, gap, rows, columns, itemCount, overscan]);

  // Get position for an item
  const getItemPosition = useCallback((index: number) => {
    const row = Math.floor(index / columns);
    const col = index % columns;
    
    return {
      top: row * (itemHeight + gap),
      left: col * (itemWidth + gap),
      width: itemWidth,
      height: itemHeight,
    };
  }, [columns, itemWidth, itemHeight, gap]);

  // Handle scroll
  const handleScroll = useCallback((e: React.UIEvent<HTMLDivElement>) => {
    setScrollTop(e.currentTarget.scrollTop);
  }, []);

  // Handle container resize
  useEffect(() => {
    const container = containerRef.current;
    if (!container) return;

    const resizeObserver = new ResizeObserver((entries) => {
      for (const entry of entries) {
        setContainerWidth(entry.contentRect.width);
      }
    });

    resizeObserver.observe(container);
    setContainerWidth(container.clientWidth);

    return () => resizeObserver.disconnect();
  }, []);

  return {
    containerRef,
    scrollTop,
    columns,
    rows,
    totalHeight,
    ...visibleRange,
    getItemPosition,
    handleScroll,
    containerWidth,
  };
}

export default useVirtualGrid;

const GAP = 16;
const OVERSCAN = 2;

export const AssetGrid: React.FC<AssetGridProps> = ({
  files,
  selectedIds,
  onSelect,
  onDoubleClick,
  thumbnailSize = 256,
  className,
}) => {
  const { ensureThumbnail, isLoading, thumbnails } = useThumbnails();
  const [lastClicked, setLastClicked] = useState<number | null>(null);
  
  const {

      }
    } else {
      onSelect(file, isMulti);
      setLastClicked(file.id);
    }
  }, [files, lastClicked, onSelect]);

  if (files.length === 0) {
    return (
      <div className={`flex flex-col items-center justify-center h-full text-gray-500 ${className || ''}`}>
        <ImageIcon className="w-16 h-16 mb-4 opacity-50" />
        <p className="text-sm">No images in current folder</p>
      </div>
    );
  }

  return (
    <div
      ref={containerRef}
      className={`relative overflow-auto h-full ${className || ''}`}
      onScroll={handleScroll}
    >
      <div style={{ height: totalHeight, position: 'relative' }}>
        {visibleItems.map(index => {
          const file = files[index];
          if (!file) return null;
          
          const position = getItemPosition(index);
          const thumbKey = `${file.id}_${thumbnailSize}`;
          const thumbnailUrl = thumbnails.get(thumbKey);
          const loading = isLoading(file.id, thumbnailSize);
          const isSelected = selectedIds.has(file.id);
          
          return (
            <AssetThumbnail
              key={file.id}
              file={file}
              thumbnailUrl={thumbnailUrl}
              isLoading={loading}
              isSelected={isSelected}
              onClick={() => handleClick(file, event as unknown as React.MouseEvent)}
              onDoubleClick={() => onDoubleClick(file)}
              style={{
                position: 'absolute',
                top: position.top,
                left: position.left,
                width: position.width,
                height: position.height,
              }}
            />
          );
        })}
      </div>
    </div>
  );
};

export default AssetGrid;

impl Database {
    pub async fn get_all_tags(&self) -> Result<Vec<Tag>, sqlx::Error> {
        sqlx::query_as::<_, (i64, String, String)>(
            "SELECT id, name, color FROM tags ORDER BY name"
        )
        .fetch_all(&self.pool)
        .await
        .map(|rows| {
            rows.into_iter()
                .map(|(id, name, color)| Tag { id, name, color })
                .collect()
        })
    }
    
    pub async fn create_tag(&self, name: &str, color: &str) -> Result<Tag, sqlx::Error> {
        let id = sqlx::query(
            "INSERT INTO tags (name, color) VALUES (?, ?)"
        )
        .bind(name)
        .bind(color)
        .execute(&self.pool)
        .await?
        .last_insert_rowid();
        
        Ok(Tag {
            id,
            name: name.to_string(),
            color: color.to_string(),
        })
    }
    
    pub async fn delete_tag(&self, tag_id: i64) -> Result<(), sqlx::Error> {
        sqlx::query("DELETE FROM tags WHERE id = ?")
            .bind(tag_id)
            .execute(&self.pool)
            .await?;
        Ok(())
    }
    
    pub async fn get_file_tags(&self, file_id: i64) -> Result<Vec<Tag>, sqlx::Error> {
        sqlx::query_as::<_, (i64, String, String)>(
            r#"SELECT t.id, t.name, t.color 
               FROM tags t
               JOIN file_tags ft ON t.id = ft.tag_id
               WHERE ft.file_id = ?
               ORDER BY t.name"#
        )
        .bind(file_id)
        .fetch_all(&self.pool)
        .await
        .map(|rows| {
            rows.into_iter()
                .map(|(id, name, color)| Tag { id, name, color })
                .collect()
        })
    }
    
    pub async fn add_tag_to_file(&self, file_id: i64, tag_id: i64) -> Result<(), sqlx::Error> {
        sqlx::query(
            "INSERT OR IGNORE INTO file_tags (file_id, tag_id) VALUES (?, ?)"
        )
        .bind(file_id)
        .bind(tag_id)
        .execute(&self.pool)
        .await?;
        Ok(())
    }
    
    pub async fn remove_tag_from_file(&self, file_id: i64, tag_id: i64) -> Result<(), sqlx::Error> {
        sqlx::query(
            "DELETE FROM file_tags WHERE file_id = ? AND tag_id = ?"
        )
        .bind(file_id)
        .bind(tag_id)
        .execute(&self.pool)
        .await?;
        Ok(())
    }
}
