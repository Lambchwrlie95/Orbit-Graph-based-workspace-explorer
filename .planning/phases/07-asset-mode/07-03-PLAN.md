**From Phase 7 Context:**
Perceptual Hash (pHash) Algorithm:
1. Resize image to 32x32
2. Convert to grayscale
3. Apply DCT (Discrete Cosine Transform)
4. Take top-left 8x8 frequencies (low frequencies)
5. Compare to median, create 64-bit hash
6. Hamming distance between hashes determines similarity

Similarity Thresholds:
- Distance 0-5: Exact/near-exact match
- Distance 6-10: Very similar
- Distance 11-20: Similar composition

**pHash Implementation Notes:**
- Use DCT (Discrete Cosine Transform) from rustdct crate or implement simple version
- 64-bit hash stored as 8 bytes in SQLite BLOB
- Hamming distance = count of differing bits
- Group by threshold: exact (≤5), similar (≤15) configurable

use image::{DynamicImage, GenericImageView, GrayImage, Luma};
use rustdct::DctPlanner;
use serde::{Deserialize, Serialize};
use std::path::Path;

const HASH_SIZE: usize = 8;
const DCT_SIZE: usize = 32;

#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
pub struct PerceptualHash {
    pub hash: u64,
}

impl PerceptualHash {
    pub fn from_bytes(bytes: &[u8]) -> Option<Self> {
        if bytes.len() >= 8 {
            let mut arr = [0u8; 8];
            arr.copy_from_slice(&bytes[..8]);
            Some(Self {
                hash: u64::from_be_bytes(arr),
            })
        } else {
            None
        }
    }

    pub fn to_bytes(&self) -> Vec<u8> {
        self.hash.to_be_bytes().to_vec()
    }

    pub fn hamming_distance(&self, other: &PerceptualHash) -> u32 {
        (self.hash ^ other.hash).count_ones()
    }
}

pub fn generate_perceptual_hash(path: &Path) -> Result<PerceptualHash, String> {
    // Load image
    let img = image::open(path).map_err(|e| e.to_string())?;
    
    // Resize to 32x32 and convert to grayscale
    let small = img.resize_exact(DCT_SIZE as u32, DCT_SIZE as u32, image::imageops::FilterType::Lanczos3);
    let gray = small.to_luma8();
    
    // Apply DCT
    let dct_coeffs = apply_dct(&gray);
    
    // Take top-left 8x8 of low frequencies (skip DC component at 0,0)
    let mut low_freqs = Vec::with_capacity(HASH_SIZE * HASH_SIZE);
    for y in 0..HASH_SIZE {
        for x in 0..HASH_SIZE {
            if x == 0 && y == 0 {
                continue; // Skip DC component
            }
            low_freqs.push(dct_coeffs[y * DCT_SIZE + x]);
        }
    }
    
    // Calculate median
    let mut sorted = low_freqs.clone();
    sorted.sort_by(|a, b| a.partial_cmp(b).unwrap());
    let median = sorted[sorted.len() / 2];
    
    // Build hash: 1 if above median, 0 if below
    let mut hash: u64 = 0;
    for (i, &val) in low_freqs.iter().enumerate() {
        if val > median {
            hash |= 1 << i;
        }
    }
    
    Ok(PerceptualHash { hash })
}

fn apply_dct(gray: &GrayImage) -> Vec<f32> {
    let size = DCT_SIZE;
    let mut planner = DctPlanner::new();
    let dct = planner.plan_dct2(size);
    
    // Convert image to f32 vector (row-major)
    let mut data: Vec<f32> = gray.pixels()
        .map(|p| p[0] as f32)
        .collect();
    
    // Apply DCT to rows
    let mut row_dct = vec![0.0f32; size * size];
    for y in 0..size {
        let mut row: Vec<f32> = (0..size).map(|x| data[y * size + x]).collect();
        dct.process_dct2(&mut row);
        for x in 0..size {
            row_dct[y * size + x] = row[x];
        }
    }
    
    // Apply DCT to columns
    let mut result = vec![0.0f32; size * size];
    for x in 0..size {
        let mut col: Vec<f32> = (0..size).map(|y| row_dct[y * size + x]).collect();
        dct.process_dct2(&mut col);
        for y in 0..size {
            result[y * size + x] = col[y];
        }
    }
    
    result
}

pub fn hamming_distance(hash1: &PerceptualHash, hash2: &PerceptualHash) -> u32 {
    hash1.hamming_distance(hash2)
}

/// Group hashes by similarity threshold
pub fn group_similar_hashes(
    hashes: &[(i64, PerceptualHash)],
    threshold: u32,
) -> Vec<Vec<i64>> {
    let mut groups: Vec<Vec<i64>> = Vec::new();
    let mut assigned: std::collections::HashSet<i64> = std::collections::HashSet::new();
    
    for (i, (id1, hash1)) in hashes.iter().enumerate() {
        if assigned.contains(id1) {
            continue;
        }
        
        let mut group = vec![*id1];
        assigned.insert(*id1);
        
        for (id2, hash2) in hashes.iter().skip(i + 1) {
            if assigned.contains(id2) {
                continue;
            }
            
            if hamming_distance(hash1, hash2) <= threshold {
                group.push(*id2);
                assigned.insert(*id2);
            }
        }
        
        if group.len() > 1 {
            groups.push(group);
        }
    }
    
    groups
}

#[derive(Debug, Serialize, Deserialize)]
pub struct DuplicateGroup {
    pub group_id: usize,
    pub files: Vec<DuplicateFileInfo>,
    pub similarity_type: SimilarityType,
    pub wasted_space: u64,
}

#[derive(Debug, Serialize, Deserialize)]
pub enum SimilarityType {
    Exact,      // Hamming distance 0
    VerySimilar, // Hamming distance 1-5
    Similar,    // Hamming distance 6-15
}

#[derive(Debug, Serialize, Deserialize)]
pub struct DuplicateFileInfo {
    pub file_id: i64,
    pub path: String,
    pub name: String,
    pub size: u64,
    pub thumbnail_path: Option<String>,
    pub is_original: bool, // First in group = original
}

#[derive(Debug, Serialize, Deserialize)]
pub struct FindDuplicatesRequest {
    pub path: Option<String>, // Specific folder or None for all
    pub threshold: u32,       // Max Hamming distance (0-64)
    pub min_group_size: usize,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct SimilarImage {
    pub file_id: i64,
    pub path: String,
    pub name: String,
    pub distance: u32,
    pub thumbnail_path: Option<String>,
}

#[tauri::command]
pub async fn find_duplicate_groups(
    request: FindDuplicatesRequest,
    db: State<'_, Arc<Database>>,
) -> Result<Vec<DuplicateGroup>, String> {
    let threshold = request.threshold.clamp(0, 64);
    
    // Get all images with perceptual hashes
    let hashes = if let Some(path) = request.path {
        db.get_hashes_in_folder(&path).await.map_err(|e| e.to_string())?
    } else {
        db.get_all_hashes().await.map_err(|e| e.to_string())?
    };
    
    if hashes.is_empty() {
        return Ok(vec![]);
    }
    
    // Group by similarity
    let file_ids: Vec<i64> = hashes.iter().map(|(id, _)| *id).collect();
    let hash_values: Vec<PerceptualHash> = hashes.into_iter()
        .map(|(_, bytes)| PerceptualHash::from_bytes(&bytes).ok_or_else(|| "Invalid hash".to_string()))
        .collect::<Result<Vec<_>, _>>()?;
    
    let hash_pairs: Vec<(i64, PerceptualHash)> = file_ids.into_iter()
        .zip(hash_values.into_iter())
        .collect();
    
    let groups = group_similar_hashes(&hash_pairs, threshold);
    
    // Build duplicate group info
    let mut result = Vec::new();
    for (group_id, file_ids) in groups.iter().enumerate() {
        if file_ids.len() < request.min_group_size {
            continue;
        }
        
        let mut files = Vec::new();
        let mut wasted_space: u64 = 0;
        
        for (i, file_id) in file_ids.iter().enumerate() {
            if let Ok(Some(file)) = db.get_file_by_id(*file_id).await {
                // Get thumbnail path
                let thumbnail = db.get_thumbnail(*file_id, 256).await.ok().flatten();
                
                files.push(DuplicateFileInfo {
                    file_id: *file_id,
                    path: file.path.clone(),
                    name: file.name.clone(),
                    size: file.size as u64,
                    thumbnail_path: thumbnail.map(|t| t.path),
                    is_original: i == 0,
                });
                
                if i > 0 {
                    wasted_space += file.size as u64;
                }
            }
        }
        
        // Determine similarity type based on threshold
        let similarity_type = if threshold <= 5 {
            SimilarityType::Exact
        } else if threshold <= 10 {
            SimilarityType::VerySimilar
        } else {
            SimilarityType::Similar
        };
        
        result.push(DuplicateGroup {
            group_id,
            files,
            similarity_type,
            wasted_space,
        });
    }
    
    // Sort by wasted space (descending)
    result.sort_by(|a, b| b.wasted_space.cmp(&a.wasted_space));
    
    Ok(result)
}

#[tauri::command]
pub async fn get_similar_images(
    file_id: i64,
    threshold: u32,
    db: State<'_, Arc<Database>>,
) -> Result<Vec<SimilarImage>, String> {
    let threshold = threshold.clamp(0, 64);
    
    // Get target file hash
    let target_hash = db.get_hash_for_file(file_id)
        .await
        .map_err(|e| e.to_string())?
        .ok_or_else(|| "File has no perceptual hash".to_string())?;
    
    let target = PerceptualHash::from_bytes(&target_hash)
        .ok_or_else(|| "Invalid hash format".to_string())?;
    
    // Get all other hashes
    let all_hashes = db.get_all_hashes().await.map_err(|e| e.to_string())?;
    
    let mut similar = Vec::new();
    for (other_id, other_hash_bytes) in all_hashes {
        if other_id == file_id {
            continue;
        }
        
        if let Some(other) = PerceptualHash::from_bytes(&other_hash_bytes) {
            let distance = hamming_distance(&target, &other);
            if distance <= threshold {
                if let Ok(Some(file)) = db.get_file_by_id(other_id).await {
                    let thumbnail = db.get_thumbnail(other_id, 256).await.ok().flatten();
                    
                    similar.push(SimilarImage {
                        file_id: other_id,
                        path: file.path,
                        name: file.name,
                        distance,
                        thumbnail_path: thumbnail.map(|t| t.path),
                    });
                }
            }
        }
    }
    
    // Sort by distance (most similar first)
    similar.sort_by(|a, b| a.distance.cmp(&b.distance));
    
    Ok(similar)
}

#[tauri::command]
pub async fn generate_missing_hashes(
    db: State<'_, Arc<Database>>,
) -> Result<usize, String> {
    // Find all images without perceptual hashes
    let missing = db.get_files_without_hashes().await.map_err(|e| e.to_string())?;
    let count = missing.len();
    
    for file in missing {
        let path = Path::new(&file.path);
        match generate_perceptual_hash(path) {
            Ok(hash) => {
                let _ = db.store_perceptual_hash(file.id, &hash.to_bytes()).await;
            }
            Err(e) => {
                eprintln!("Failed to hash {}: {}", file.path, e);
            }
        }
    }
    
    Ok(count)
}

#[tauri::command]
pub async fn delete_duplicate(

    db: State<'_, Arc<Database>>,

impl Database {
    pub async fn get_all_hashes(&self) -> Result<Vec<(i64, Vec<u8>)>, sqlx::Error> {
        sqlx::query_as::<_, (i64, Vec<u8>)>(
            "SELECT file_id, phash FROM perceptual_hashes"
        )
        .fetch_all(&self.pool)
        .await
    }
    
    pub async fn get_hashes_in_folder(&self, path: &str) -> Result<Vec<(i64, Vec<u8>)>, sqlx::Error> {
        sqlx::query_as::<_, (i64, Vec<u8>)>(
            r#"SELECT ph.file_id, ph.phash 
               FROM perceptual_hashes ph
               JOIN files f ON ph.file_id = f.id
               WHERE f.path LIKE ?"#
        )
        .bind(format!("{}%", path))
        .fetch_all(&self.pool)
        .await
    }
    
    pub async fn get_hash_for_file(&self, file_id: i64) -> Result<Option<Vec<u8>>, sqlx::Error> {
        sqlx::query_as::<_, (Vec<u8>,)>(
            "SELECT phash FROM perceptual_hashes WHERE file_id = ?"
        )
        .bind(file_id)
        .fetch_optional(&self.pool)
        .await
        .map(|opt| opt.map(|(bytes,)| bytes))
    }
    
    pub async fn store_perceptual_hash(
        &self,
        file_id: i64,
        hash: &[u8],
    ) -> Result<(), sqlx::Error> {
        sqlx::query(
            "INSERT OR REPLACE INTO perceptual_hashes (file_id, phash) VALUES (?, ?)"
        )
        .bind(file_id)
        .bind(hash)
        .execute(&self.pool)
        .await?;
        Ok(())
    }
    
    pub async fn get_files_without_hashes(&self) -> Result<Vec<FileRecord>, sqlx::Error> {
        sqlx::query_as::<_, FileRecord>(
            r#"SELECT f.* FROM files f
               LEFT JOIN perceptual_hashes ph ON f.id = ph.file_id
               WHERE ph.file_id IS NULL
               AND f.extension IN ('jpg', 'jpeg', 'png', 'gif', 'webp', 'bmp')"#
        )
        .fetch_all(&self.pool)
        .await
    }
    
    pub async fn count_perceptual_hashes(&self) -> Result<i64, sqlx::Error> {
        sqlx::query_scalar("SELECT COUNT(*) FROM perceptual_hashes")
            .fetch_one(&self.pool)
            .await
    }
    
    pub async fn count_image_files(&self) -> Result<i64, sqlx::Error> {
        sqlx::query_scalar(
            "SELECT COUNT(*) FROM files WHERE extension IN ('jpg', 'jpeg', 'png', 'gif', 'webp', 'bmp')"
        )
        .fetch_one(&self.pool)
        .await
    }
    
    pub async fn delete_file(&self, file_id: i64) -> Result<(), sqlx::Error> {
        sqlx::query("DELETE FROM files WHERE id = ?")
            .bind(file_id)
            .execute(&self.pool)
            .await?;
        Ok(())
    }
}

      return await invoke<SimilarImage[]>('get_similar_images', {
        fileId,
        threshold,
      });
    } catch (err) {
      console.error('Failed to get similar images:', err);
      return [];
    }
  }, []);

  // Generate missing perceptual hashes
  const generateMissingHashes = useCallback(async (): Promise<number> => {
    setScanning(true);
    setError(null);
    
    try {
      const count = await invoke<number>('generate_missing_hashes');
      await refreshStats();
      return count;
    } catch (err) {
      const message = err instanceof Error ? err.message : String(err);
      setError(message);
      return 0;
    } finally {
      setScanning(false);
    }
  }, []);

  // Delete a duplicate file
  const deleteDuplicate = useCallback(async (fileId: number): Promise<boolean> => {
    try {
      await invoke('delete_duplicate', { fileId });
      // Remove from groups
      setGroups(prev => 
        prev.map(group => ({
          ...group,
          files: group.files.filter(f => f.file_id !== fileId),
        })).filter(group => group.files.length > 1)
      );
      return true;
    } catch (err) {
      console.error('Failed to delete duplicate:', err);
      return false;
    }
  }, []);

  // Get duplicate statistics
  const getStats = useCallback(async (): Promise<DuplicateStats | null> => {
    try {
      const result = await invoke<DuplicateStats>('get_duplicate_stats');
      setStats(result);
      return result;
    } catch (err) {
      console.error('Failed to get stats:', err);
      return null;
    }
  }, []);

  const refreshStats = useCallback(async () => {
    await getStats();
  }, [getStats]);

  // Format bytes for display
  const formatWastedSpace = useCallback((bytes: number): string => {
    if (bytes === 0) return '0 B';
    const k = 1024;
    const sizes = ['B', 'KB', 'MB', 'GB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    return `${(bytes / Math.pow(k, i)).toFixed(1)} ${sizes[i]}`;
  }, []);

  // Get similarity label
  const getSimilarityLabel = useCallback((type: SimilarityType): string => {
    switch (type) {
      case 'Exact': return 'Exact Duplicate';
      case 'VerySimilar': return 'Very Similar';
      case 'Similar': return 'Similar';
      default: return 'Unknown';
    }
  }, []);

  // Load stats on mount
  useEffect(() => {
    getStats();
  }, [getStats]);

  return {
    groups,
    loading,
    scanning,
    stats,
    error,
    findDuplicates,
    getSimilarImages,
    generateMissingHashes,
    deleteDuplicate,
    refreshStats,
    formatWastedSpace,
    getSimilarityLabel,
  };
}

export default useDuplicateDetection;
