```sql
-- Thumbnail metadata storage
CREATE TABLE IF NOT EXISTS thumbnails (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    file_id INTEGER NOT NULL REFERENCES files(id) ON DELETE CASCADE,
    size INTEGER NOT NULL, -- 128, 256, or 512
    path TEXT NOT NULL, -- relative path from thumbnail base dir
    generated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    file_modified_at TIMESTAMP, -- for cache invalidation
    width INTEGER, -- actual thumbnail width
    height INTEGER, -- actual thumbnail height
    UNIQUE(file_id, size)
);

-- Index for fast lookups
CREATE INDEX IF NOT EXISTS idx_thumbnails_file_id ON thumbnails(file_id);
CREATE INDEX IF NOT EXISTS idx_thumbnails_size ON thumbnails(size);

-- Perceptual hash table (for 07-03)
CREATE TABLE IF NOT EXISTS perceptual_hashes (
    file_id INTEGER PRIMARY KEY REFERENCES files(id) ON DELETE CASCADE,
    phash BLOB NOT NULL, -- 64-bit hash stored as 8 bytes
    algorithm TEXT DEFAULT 'phash',
    generated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Asset tags tables (for 07-02)
CREATE TABLE IF NOT EXISTS tags (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL UNIQUE,
    color TEXT DEFAULT '#6366F1', -- default indigo
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS file_tags (
    file_id INTEGER NOT NULL REFERENCES files(id) ON DELETE CASCADE,
    tag_id INTEGER NOT NULL REFERENCES tags(id) ON DELETE CASCADE,
    PRIMARY KEY (file_id, tag_id)
);

CREATE INDEX IF NOT EXISTS idx_file_tags_tag_id ON file_tags(tag_id);
```

```rust
use image::{imageops::FilterType, DynamicImage, ImageFormat};
use std::path::{Path, PathBuf};
use tokio::fs;
use crate::db::Database;
use crate::models::FileRecord;
use std::sync::Arc;
use tokio::sync::Semaphore;
use sha2::{Sha256, Digest};

pub const THUMBNAIL_SIZES: [u32; 3] = [128, 256, 512];
pub const THUMBNAIL_DIR: &str = "~/.local/share/orbit/thumbnails";

pub struct ThumbnailGenerator {
    db: Arc<Database>,
    semaphore: Arc<Semaphore>,
    base_path: PathBuf,
}

impl ThumbnailGenerator {
    pub fn new(db: Arc<Database>) -> Self {
        let base_path = dirs::data_local_dir()
            .unwrap_or_else(|| PathBuf::from("."))
            .join("orbit")
            .join("thumbnails");
        
        Self {
            db,
            semaphore: Arc::new(Semaphore::new(4)), // Limit concurrent processing
            base_path,
        }
    }

    pub async fn ensure_thumbnail(
        &self,
        file: &FileRecord,
        size: u32,
    ) -> Result<PathBuf, String> {
        // Check if thumbnail exists and is valid
        if let Ok(Some(thumb)) = self.db.get_thumbnail(file.id, size).await {
            let thumb_path = self.base_path.join(&thumb.path);
            if thumb_path.exists() {
                // Check if source file has been modified
                if thumb.file_modified_at == Some(file.modified_at) {
                    return Ok(thumb_path);
                }
            }
        }

        // Generate new thumbnail
        self.generate_thumbnail(file, size).await
    }

    pub async fn generate_thumbnail(
        &self,
        file: &FileRecord,
        size: u32,
    ) -> Result<PathBuf, String> {
        let permit = self.semaphore.acquire().await.map_err(|e| e.to_string())?;
        
        let source_path = Path::new(&file.path);
        let ext = source_path
            .extension()
            .and_then(|e| e.to_str())
            .unwrap_or("jpg")
            .to_lowercase();
        
        // Create thumbnail subdirectory based on file hash
        let file_hash = format!("{:x}", Sha256::digest(file.path.as_bytes()))[..16].to_string();
        let subdir = &file_hash[..2];
        let thumb_dir = self.base_path.join(subdir);
        
        fs::create_dir_all(&thumb_dir).await.map_err(|e| e.to_string())?;
        
        let thumb_filename = format!("{}_{}.jpg", &file_hash, size);
        let thumb_path = thumb_dir.join(&thumb_filename);
        
        // Load and resize image
        let img = tokio::task::spawn_blocking({
            let path = source_path.to_path_buf();
            move || -> Result<DynamicImage, String> {
                image::open(&path).map_err(|e| e.to_string())
            }
        }).await.map_err(|e| e.to_string())??;
        
        let (orig_width, orig_height) = (img.width(), img.height());
        let thumbnail = img.resize(size, size, FilterType::Lanczos3);
        let (thumb_width, thumb_height) = (thumbnail.width(), thumbnail.height());
        
        // Save thumbnail
        tokio::task::spawn_blocking({
            let path = thumb_path.clone();
            move || -> Result<(), String> {
                thumbnail.save_with_format(&path, ImageFormat::Jpeg)
                    .map_err(|e| e.to_string())
            }
        }).await.map_err(|e| e.to_string())??;
        
        drop(permit);
        
        // Store in database
        let rel_path = Path::new(subdir).join(&thumb_filename);
        self.db.upsert_thumbnail(
            file.id,
            size as i32,
            rel_path.to_string_lossy().to_string(),
            file.modified_at,
            thumb_width as i32,
            thumb_height as i32,
        ).await.map_err(|e| e.to_string())?;
        
        Ok(thumb_path)
    }

    pub async fn delete_thumbnails_for_file(&self, file_id: i64) -> Result<(), String> {
        if let Ok(thumbs) = self.db.get_thumbnails_for_file(file_id).await {
            for thumb in thumbs {
                let path = self.base_path.join(&thumb.path);
                let _ = fs::remove_file(path).await;
            }
        }
        self.db.delete_thumbnails_for_file(file_id).await.map_err(|e| e.to_string())
    }
}
```

```rust
use tauri::State;
use crate::services::thumbnail_generator::{ThumbnailGenerator, THUMBNAIL_SIZES};
use crate::db::Database;
use std::sync::Arc;
use serde::{Deserialize, Serialize};

#[derive(Debug, Serialize, Deserialize)]
pub struct ThumbnailInfo {
    pub file_id: i64,
    pub size: u32,
    pub path: String,
    pub width: i32,
    pub height: i32,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct EnsureThumbnailRequest {
    pub file_id: i64,
    pub size: u32,
}

#[tauri::command]
pub async fn ensure_thumbnail(
    request: EnsureThumbnailRequest,
    db: State<'_, Arc<Database>>,
    generator: State<'_, Arc<ThumbnailGenerator>>,
) -> Result<String, String> {
    // Get file record from database
    let file = db.get_file_by_id(request.file_id)
        .await
        .map_err(|e| e.to_string())?
        .ok_or_else(|| "File not found".to_string())?;
    
    // Validate requested size
    if !THUMBNAIL_SIZES.contains(&request.size) {
        return Err(format!("Invalid thumbnail size: {}. Valid sizes: {:?}", 
            request.size, THUMBNAIL_SIZES));
    }
    
    // Generate or retrieve thumbnail
    let thumb_path = generator.ensure_thumbnail(&file, request.size).await?;
    
    Ok(thumb_path.to_string_lossy().to_string())
}

#[tauri::command]
pub async fn get_thumbnail_info(
    file_id: i64,
    db: State<'_, Arc<Database>>,
) -> Result<Vec<ThumbnailInfo>, String> {
    let thumbs = db.get_thumbnails_for_file(file_id)
        .await
        .map_err(|e| e.to_string())?;
    
    Ok(thumbs.into_iter().map(|t| ThumbnailInfo {
        file_id: t.file_id,
        size: t.size as u32,
        path: t.path,
        width: t.width,
        height: t.height,
    }).collect())
}

#[tauri::command]
pub async fn delete_thumbnails(
    file_id: i64,
    generator: State<'_, Arc<ThumbnailGenerator>>,
) -> Result<(), String> {
    generator.delete_thumbnails_for_file(file_id).await
}

#[tauri::command]
pub fn get_supported_thumbnail_sizes() -> Vec<u32> {
    THUMBNAIL_SIZES.to_vec()
}
```

```typescript
import { useState, useCallback } from 'react';
import { invoke } from '@tauri-apps/api/core';

interface ThumbnailRequest {
  file_id: number;
  size: number;
}

interface ThumbnailInfo {
  file_id: number;
  size: number;
  path: string;
  width: number;
  height: number;
}

export function useThumbnails() {
  const [loading, setLoading] = useState<Set<string>>(new Set());
  const [thumbnails, setThumbnails] = useState<Map<string, string>>(new Map());
  const [errors, setErrors] = useState<Map<string, string>>(new Map());

  const getThumbnailKey = (fileId: number, size: number): string => 
    `${fileId}_${size}`;

  const ensureThumbnail = useCallback(async (
    fileId: number, 
    size: number
  ): Promise<string | null> => {
    const key = getThumbnailKey(fileId, size);
    
    // Return cached
    if (thumbnails.has(key)) {
      return thumbnails.get(key)!;
    }
    
    // Skip if already loading
    if (loading.has(key)) {
      return null;
    }
    
    setLoading(prev => new Set(prev).add(key));
    
    try {
      const path = await invoke<string>('ensure_thumbnail', {
        request: { file_id: fileId, size }
      });
      
      // Convert to asset URL for Tauri
      const assetUrl = `asset://localhost${path}`;
      
      setThumbnails(prev => new Map(prev).set(key, assetUrl));
      setErrors(prev => {
        const next = new Map(prev);
        next.delete(key);
        return next;
      });
      
      return assetUrl;
    } catch (err) {
      const errorMsg = err instanceof Error ? err.message : String(err);
      setErrors(prev => new Map(prev).set(key, errorMsg));
      return null;
    } finally {
      setLoading(prev => {
        const next = new Set(prev);
        next.delete(key);
        return next;
      });
    }
  }, [thumbnails, loading]);

  const getThumbnailInfo = useCallback(async (fileId: number): Promise<ThumbnailInfo[]> => {
    try {
      return await invoke<ThumbnailInfo[]>('get_thumbnail_info', { fileId });
    } catch (err) {
      console.error('Failed to get thumbnail info:', err);
      return [];
    }
  }, []);

  const isLoading = useCallback((fileId: number, size: number): boolean => {
    return loading.has(getThumbnailKey(fileId, size));
  }, [loading]);

  const getError = useCallback((fileId: number, size: number): string | undefined => {
    return errors.get(getThumbnailKey(fileId, size));
  }, [errors]);

  return {
    thumbnails,
    ensureThumbnail,
    getThumbnailInfo,
    isLoading,
    getError,
    loadingCount: loading.size,
  };
}
```
