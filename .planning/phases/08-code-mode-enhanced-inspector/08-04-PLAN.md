### Database Tables (from 07-01)
- `thumbnails` table: Stores thumbnail paths for images
- `image_metadata` table: Stores dimensions, format, file size
- `perceptual_hashes` table: Stores pHash for duplicate detection

### Backend Services (from 07-01, 07-02, 07-03)
- `ThumbnailGenerator`: Generates thumbnails at 128/256/512px
- `ColorExtractor`: Extracts dominant colors using k-means
- `DuplicateDetector`: Generates pHash and finds similar images

### Existing Data
When Phase 7 is complete, images should have:
- Thumbnails generated (multiple sizes)
- Dominant colors extracted (4-6 colors)
- Perceptual hashes computed
- Similar images identified

This plan CONSUMES that data for the inspector, it does not recreate it.

## Technical Requirements

### Required Queries
1. Get image dimensions from image_metadata table
2. Get dominant colors from color extraction (stored in image_metadata as JSON or separate table)
3. Get similar images by querying perceptual_hashes table for Hamming distance < threshold
4. Get asset references by searching markdown_links and code files for this image path

### Inspector Display
Dimensions, colors as swatches, similar image thumbnails, referencing files list.

```typescript
import { useState, useEffect, useCallback } from 'react';
import { invoke } from '@tauri-apps/api/core';

interface ImageDimensions {
  width: number;
  height: number;
  aspect_ratio: string;
}

interface ColorInfo {
  hex: string;
  rgb: [number, number, number];
}

interface SimilarImage {
  file_id: number;
  file_path: string;
  similarity_score: number;
  thumbnail_path: string | null;
}

interface AssetReference {
  source_file_id: number;
  source_path: string;
  reference_type: string;
  line_number: number | null;
}

interface UseImageAnalysisReturn {
  dimensions: ImageDimensions | null;
  colors: ColorInfo[];
  similarImages: SimilarImage[];
  references: AssetReference[];
  loading: boolean;
  error: string | null;
  refresh: () => void;
}

export function useImageAnalysis(fileId: number | null): UseImageAnalysisReturn {
  const [dimensions, setDimensions] = useState<ImageDimensions | null>(null);
  const [colors, setColors] = useState<ColorInfo[]>([]);
  const [similarImages, setSimilarImages] = useState<SimilarImage[]>([]);
  const [references, setReferences] = useState<AssetReference[]>([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const fetchAnalysis = useCallback(async () => {
    if (fileId === null) return;
    
    setLoading(true);
    setError(null);
    
    try {
      // Fetch all data in parallel
      const [dims, cols, similar, refs] = await Promise.all([
        invoke<ImageDimensions | null>('get_image_dimensions', { fileId }),
        invoke<ColorInfo[]>('get_image_colors', { fileId }),
        invoke<SimilarImage[]>('get_similar_images', { fileId, threshold: 10 }),
        invoke<AssetReference[]>('get_asset_references', { assetFileId: fileId }),
      ]);
      
      setDimensions(dims);
      setColors(cols);
      setSimilarImages(similar);
      setReferences(refs);
    } catch (err) {
      setError(err instanceof Error ? err.message : String(err));
    } finally {
      setLoading(false);
    }
  }, [fileId]);

  useEffect(() => {
    fetchAnalysis();
  }, [fetchAnalysis]);

  return {
    dimensions,
    colors,
    similarImages,
    references,
    loading,
    error,
    refresh: fetchAnalysis,
  };
}
```
