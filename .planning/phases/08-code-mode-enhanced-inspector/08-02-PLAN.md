```rust
use git2::{Repository, Status, StatusOptions, Error};
use std::path::Path;
use serde::{Deserialize, Serialize};

#[derive(Debug, Serialize, Deserialize)]
pub struct FileGitStatus {
    pub status: GitStatus,
    pub additions: Option<u32>,
    pub deletions: Option<u32>,
}

#[derive(Debug, Serialize, Deserialize)]
#[serde(rename_all = "snake_case")]
pub enum GitStatus {
    Current,           // No changes
    Modified,          // Modified but not staged
    Staged,            // Staged for commit
    StagedModified,    // Staged with further modifications
    New,               // Untracked file
    Deleted,           // Deleted
    Renamed,           // Renamed
    Ignored,           // In .gitignore
    Conflicted,        // Merge conflict
    Unknown,           // Error or not in git repo
}

impl From<Status> for GitStatus {
    fn from(status: Status) -> Self {
        use Status::*;
        match status {
            s if s.is_conflicted() => GitStatus::Conflicted,
            s if s.is_ignored() => GitStatus::Ignored,
            s if s.is_index_new() || s.is_wt_new() => GitStatus::New,
            s if s.is_index_deleted() || s.is_wt_deleted() => GitStatus::Deleted,
            s if s.is_index_renamed() || s.is_wt_renamed() => GitStatus::Renamed,
            s if s.is_index_modified() && s.is_wt_modified() => GitStatus::StagedModified,
            s if s.is_index_modified() => GitStatus::Staged,
            s if s.is_wt_modified() => GitStatus::Modified,
            _ => GitStatus::Current,
        }
    }
}

pub fn get_file_status(repo_path: &Path, file_path: &Path) -> Option<FileGitStatus> {
    let repo = Repository::open(repo_path).ok()?;
    let relative_path = file_path.strip_prefix(repo_path).ok()?;
    
    let mut opts = StatusOptions::new();
    opts.include_untracked(true)
        .include_ignored(false)
        .pathspec(relative_path.to_str()?);
    
    let statuses = repo.status_file(relative_path).ok()?;
    
    Some(FileGitStatus {
        status: statuses.into(),
        additions: None,  // Could use git2::Diff for line counts
        deletions: None,
    })
}

pub fn find_repo_root(file_path: &Path) -> Option<&Path> {
    // Walk up directory tree to find .git directory
    let mut current = file_path.parent()?;
    loop {
        if current.join(".git").exists() {
            return Some(current);
        }
        current = current.parent()?;
    }
}
```

Hook interface:
```typescript
interface UseCodeAnalysisReturn {
  analysis: CodeAnalysis | null;
  gitStatus: FileGitStatus | null;
  relatedFiles: string[];
  loading: boolean;
  error: string | null;
  refresh: () => void;
}

interface CodeAnalysis {
  imports: Array<{
    name: string;
    path: string;
    import_type: 'local' | 'package' | 'std';
  }>;
  exports: Array<{
    name: string;
    export_type: string;
  }>;
}

interface FileGitStatus {
  status: 'current' | 'modified' | 'staged' | 'staged_modified' | 'new' | 'deleted' | 'renamed' | 'ignored' | 'conflicted' | 'unknown';
  additions?: number;
  deletions?: number;
}

export function useCodeAnalysis(filePath: string | null): UseCodeAnalysisReturn {
  const [analysis, setAnalysis] = useState<CodeAnalysis | null>(null);
  const [gitStatus, setGitStatus] = useState<FileGitStatus | null>(null);
  const [relatedFiles, setRelatedFiles] = useState<string[]>([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const fetchAnalysis = useCallback(async () => {
    if (!filePath) return;
    
    setLoading(true);
    setError(null);
    
    try {
      // Fetch code analysis
      const analysisResult = await invoke<CodeAnalysis | null>('analyze_code_file', { path: filePath });
      setAnalysis(analysisResult);
      
      // Fetch git status
      try {
        const gitResult = await invoke<FileGitStatus>('get_file_git_status', { path: filePath });
        setGitStatus(gitResult);
      } catch (gitErr) {
        // Git status is optional - file might not be in a repo
        setGitStatus({ status: 'unknown' });
      }
      
      // Fetch related files
      const related = await invoke<string[]>('get_related_files', { path: filePath });
      setRelatedFiles(related);
    } catch (err) {
      setError(err instanceof Error ? err.message : String(err));
    } finally {
      setLoading(false);
    }
  }, [filePath]);

  useEffect(() => {
    fetchAnalysis();
  }, [fetchAnalysis]);

  return {
    analysis,
    gitStatus,
    relatedFiles,
    loading,
    error,
    refresh: fetchAnalysis,
  };
}
```
