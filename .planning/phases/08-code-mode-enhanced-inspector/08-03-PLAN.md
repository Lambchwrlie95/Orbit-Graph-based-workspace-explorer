use regex::Regex;
use serde::{Deserialize, Serialize};

#[derive(Debug, Serialize, Deserialize)]
pub struct MarkdownAnalysis {
    pub headings: Vec<Heading>,
    pub links: Vec<Link>,
    pub tags: Vec<Tag>,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct Heading {
    pub level: u8,      // 1-6
    pub text: String,
    pub line_number: usize,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct Link {
    pub text: String,
    pub target: String,
    pub link_type: LinkType,
    pub line_number: usize,
    pub is_valid: Option<bool>,  // Validated later
}

#[derive(Debug, Serialize, Deserialize)]
#[serde(rename_all = "snake_case")]
pub enum LinkType {
    Markdown,   // [text](path)
    Wiki,       // [[path]]
    Url,        // [text](http://...)
}

#[derive(Debug, Serialize, Deserialize)]
pub struct Tag {
    pub name: String,
    pub line_number: usize,
}

pub fn analyze_markdown(content: &str) -> MarkdownAnalysis {
    let headings = extract_headings(content);
    let links = extract_links(content);
    let tags = extract_tags(content);
    
    MarkdownAnalysis {
        headings,
        links,
        tags,
    }
}

fn extract_headings(content: &str) -> Vec<Heading> {
    let re = Regex::new(r"^(#{1,6})\s+(.+)$").unwrap();
    content
        .lines()
        .enumerate()
        .filter_map(|(i, line)| {
            re.captures(line).map(|cap| {
                let level = cap[1].len() as u8;
                let text = cap[2].trim().to_string();
                Heading {
                    level,
                    text,
                    line_number: i + 1,
                }
            })
        })
        .collect()
}

fn extract_links(content: &str) -> Vec<Link> {
    let mut links = Vec::new();
    
    // Standard markdown links
    let md_re = Regex::new(r"\[([^\]]+)\]\(([^)]+)\)").unwrap();
    for (i, line) in content.lines().enumerate() {
        for cap in md_re.captures_iter(line) {
            let text = cap[1].to_string();
            let target = cap[2].to_string();
            let link_type = if target.starts_with("http://") || target.starts_with("https://") {
                LinkType::Url
            } else {
                LinkType::Markdown
            };
            links.push(Link {
                text,
                target,
                link_type,
                line_number: i + 1,
                is_valid: None,
            });
        }
    }
    
    // Wiki-style links [[Target]] or [[Target|Text]]
    let wiki_re = Regex::new(r"\[\[([^\]|]+)(?:\|([^\]]+))?\]\]").unwrap();
    for (i, line) in content.lines().enumerate() {
        for cap in wiki_re.captures_iter(line) {
            let target = cap[1].to_string();
            let text = cap.get(2).map(|m| m.as_str().to_string())
                .unwrap_or_else(|| target.clone());
            links.push(Link {
                text,
                target,
                link_type: LinkType::Wiki,
                line_number: i + 1,
                is_valid: None,
            });
        }
    }
    
    links
}

fn extract_tags(content: &str) -> Vec<Tag> {
    let re = Regex::new(r"#(\w+)").unwrap();
    content
        .lines()
        .enumerate()
        .flat_map(|(i, line)| {
            re.captures_iter(line).map(move |cap| {
                Tag {
                    name: cap[1].to_string(),
                    line_number: i + 1,
                }
            })
        })
        .collect()
}

pub fn validate_links(links: &mut [Link], base_path: &std::path::Path) {
    for link in links.iter_mut() {
        match link.link_type {
            LinkType::Markdown | LinkType::Wiki => {
                // Resolve relative path
                let target_path = base_path.join(&link.target);
                link.is_valid = Some(target_path.exists());
            }
            LinkType::Url => {
                // URLs are always considered valid (we don't check online)
                link.is_valid = Some(true);
            }
        }
    }
}

import { useState, useEffect, useCallback } from 'react';
import { invoke } from '@tauri-apps/api/core';

interface Heading {
  level: number;
  text: string;
  line_number: number;
}

interface Link {
  text: string;
  target: string;
  link_type: 'markdown' | 'wiki' | 'url';
  line_number: number;
  is_valid: boolean | null;
}

interface Tag {
  name: string;
  line_number: number;
}

interface Backlink {
  source_path: string;
  link_text: string;
  line_number: number;
}

interface MarkdownAnalysis {
  headings: Heading[];
  links: Link[];
  tags: Tag[];
}

interface UseMarkdownAnalysisReturn {
  analysis: MarkdownAnalysis | null;
  backlinks: Backlink[];
  loading: boolean;
  error: string | null;
  refresh: () => void;
}

export function useMarkdownAnalysis(filePath: string | null): UseMarkdownAnalysisReturn {
  const [analysis, setAnalysis] = useState<MarkdownAnalysis | null>(null);
  const [backlinks, setBacklinks] = useState<Backlink[]>([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const fetchAnalysis = useCallback(async () => {
    if (!filePath) return;
    
    setLoading(true);
    setError(null);
    
    try {
      const result = await invoke<MarkdownAnalysis>('analyze_markdown_file', { path: filePath });
      setAnalysis(result);
      
      // Fetch backlinks
      const bl = await invoke<Backlink[]>('get_backlinks', { targetPath: filePath });
      setBacklinks(bl);
    } catch (err) {
      setError(err instanceof Error ? err.message : String(err));
    } finally {
      setLoading(false);
    }
  }, [filePath]);

  useEffect(() => {
    fetchAnalysis();
  }, [fetchAnalysis]);

  return {
    analysis,
    backlinks,
    loading,
    error,
    refresh: fetchAnalysis,
  };
}
